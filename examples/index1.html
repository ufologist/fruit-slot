调用方法¶

seajs.use(['/lottery/1.0.0/lottery','gallery/jquery/1.10.2/jquery'],function(Lottery, $){
    var lot1 = new Lottery({
        el : "#J_lottery1",
        lottoItem : '[data-lotto-item]',
        activeClass : 'lotto__item--hover'
    })
    $('#J_start1').on('click', function(e){
        e.preventDefault()
        lot1.start()
        setTimeout(function(){
            lot1.stop(2, function(){ 
                console.log(123)
            })
        },2000)
    })
})
